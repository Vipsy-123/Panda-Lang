[prog]      ->[stmts]*

[stmts]     ->[let indent =[Expr]];
            ->[exit([Expr])];

[Expr]      ->[Term]
            ->[BinExpr]

[Term]      ->[Ident]
            ->[intLit]
            ->([Expr])

[BinExpr]   ->[Expr]+[Expr]  ....prec=1
            ->[Expr]-[Expr]  ....prec=1
            ->[Expr]*[Expr]  ....prec=2
            ->[Expr]/[Expr]  ....prec=2
